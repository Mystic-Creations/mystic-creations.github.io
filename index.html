TOOLS


// Programmers

Programmers are required to use IntelliJ IDEA, either the Community Edition or Ultimate.
You can use either GitHub Desktop or Git through IntelliJ for committing changes to repositories.


// Artists

Artists can use any graphical tools they prefer. Finished art and textures are to be sent to 📑•media-and-files on Discord or commit them via GitHub Desktop.


// Writers

Writers can use any text editor they prefer, although recommended are Google Docs or Microsoft Word.
Finished documents are to be sent to ✏️•writing-and-lore on Discord.


// Translators

Translators are required to use any basic text editor along side GitHub Desktop for simplicity of work for
those who don't know how to use tools such as VS Code or IntelliJ IDEA, unless preferred otherwise

That branch will contain: README.md Main README, used for GitHub and Modrinth pages README_CURSEFORGE.md Secondary README, used for CurseForge due to different markdown support Changelogs Located at root/.changelogs/ ... PROJECT CREATION


Naming & Packages
Naming Mods, Mod IDs New project names have to be easy to remember, catchy or/and on-topic Cannot change IDs of existing projects Mod Packages Package can change if necessary on an existing project Organization-managed team project packages should look like so: net.mysticcreations.mod_id Organization-managed collaborative project packages should look like so: og_author_name.mysticcreationsteam.mod_id

Project Setup
All projects should use Architectury Loom for multi-platform support Bigger, multi-loader projects should use Architectury API as a dependency E.g. "True End", Under The Stars Smaller, multi-loader projects should avoid using Architectury API as a dependency E.g. "Create: Totem Factory", "Avaritia X Create"

Project Versioning

Our mods use versioning format: A.B.C.D
Labeled on Modrinth and Curse: A.B.C.D+mcMCVER-LOADER
e.g. 1.4.3.10+mc1.20.1-fabric

A / Content updates indicator Stays at 1 until the project no longer will recieve major content updates, only Minecraft version updates and bugfixes B / Major content updates Any versions that have added, changed, reworked, fixed or/and removed a lot of content from the mod C / Small content updates Small feature drops, changes and fixes D / Patch updates Bug fixes and small tweaks Can be overshadowed by small content updates

build.gradle 1.21.X 

plugins {
    id 'dev.architectury.loom' version '1.13-SNAPSHOT' apply false
    id 'architectury-plugin' version '3.4-SNAPSHOT'
    id 'com.gradleup.shadow' version '8.3.6' apply false
}

architectury {
    minecraft = project.minecraft_version
}

allprojects {
    group = rootProject.maven_group
    version = rootProject.mod_version //for file building
    mod_version = rootProject.mod_version
    minecraft_version = rootProject.minecraft_version
    fabric_loader_version = rootProject.fabric_loader_version
    architectury_api_version = rootProject.architectury_api_version
    mysticlib_version = rootProject.mysticlib_version

    repositories {
    }
}

def isDevBuild = (findProperty("dev_builds")?.toString()?.toBoolean() ?: false)

def gitCommitCount = providers.exec {
    commandLine "git", "rev-list", "--count", "HEAD"
}.standardOutput.asText.map { it.trim() }

subprojects {
    apply plugin: 'dev.architectury.loom'
    apply plugin: 'architectury-plugin'
    apply plugin: 'maven-publish'

    base { archivesName = rootProject.archives_name }

    repositories {
        maven {
            name = 'ParchmentMC'
            url = 'https://maven.parchmentmc.org'
        }
        maven { url = 'https://maven.minecraftforge.net/' }
        maven { url = 'https://repo.spongepowered.org/repository/maven-public/' }
        maven { 
            name = "Modrinth"
            url = uri("https://api.modrinth.com/maven")
        }
    }

    loom {
        silentMojangMappingsLicense()
    }

    dependencies {
        minecraft "net.minecraft:minecraft:$rootProject.minecraft_version"
        mappings loom.layered() {
            officialMojangMappings()
            parchment("org.parchmentmc.data:parchment-${minecraft_version}:${parchment_version}@zip")
        }
    }

    java {
        sourceCompatibility = JavaVersion.VERSION_21
        targetCompatibility = JavaVersion.VERSION_21
    }

    tasks.withType(JavaCompile).configureEach {
        it.options.release = 21
    }

    def detectedPlatform
    def name = project.name.toLowerCase()
    if (name.contains('fabric')) {
        detectedPlatform = 'Fabric'
    } else if (name.contains('neoforge')) {
        detectedPlatform = 'NeoForge'
    } else {
        detectedPlatform = 'Unknown'
    }

    if (detectedPlatform) {
        def baseVersion = "${rootProject.mod_version}+mc${rootProject.minecraft_version}-${detectedPlatform}"

        if (isDevBuild) {
            project.version = baseVersion + "-build.${gitCommitCount.get()}"
        } else {
            project.version = baseVersion
        }
    } else {
        project.version = rootProject.mod_version
    }

    tasks.withType(Jar).configureEach { Jar jarTask ->
        jarTask.archiveBaseName.set(rootProject.archives_name)
        jarTask.archiveVersion.set(project.version.toString())
    }

    tasks.matching { it.name == 'shadowJar' }.configureEach { task ->
        if (task.hasProperty('archiveBaseName')) {
            task.archiveBaseName.set(rootProject.archives_name)
            task.archiveVersion.set(project.version.toString())
        }
    }
}

Fabric build.gradle 

plugins {
    id 'com.gradleup.shadow'
}

architectury {
    platformSetupLoomIde()
    fabric()
}

configurations {
    common {
        canBeResolved = true
        canBeConsumed = false
    }
    compileClasspath.extendsFrom common
    runtimeClasspath.extendsFrom common
    developmentFabric.extendsFrom common

    shadowBundle {
        canBeResolved = true
        canBeConsumed = false
    }
}

dependencies {
    modImplementation "net.fabricmc:fabric-loader:$rootProject.fabric_loader_version"

    modImplementation "net.fabricmc.fabric-api:fabric-api:$rootProject.fabric_api_version"
    modImplementation "dev.architectury:architectury-fabric:$rootProject.architectury_api_version" //Optional

    modImplementation "maven.modrinth:mysticlib:${rootProject.mysticlib_versiob}+mc${rootProject.minecraft_version}-fabric"

    common(project(path: ':common', configuration: 'namedElements')) { transitive = false }
    shadowBundle project(path: ':common', configuration: 'transformProductionFabric')
}

processResources {
    inputs.property 'mod_version', project.mod_version
    inputs.property 'minecraft_version', project.minecraft_version
    inputs.property 'fabric_loader_version', project.fabric_loader_version
    inputs.property 'architectury_api_version', project.architectury_api_version
    inputs.property 'mysticlib_version', project.mysticlib_version

    filesMatching('fabric.mod.json') {
        expand mod_version: inputs.properties.mod_version,
                minecraft_version: inputs.properties.minecraft_version,
                architectury_api_version: inputs.properties.architectury_api_version,
                fabric_loader_version: inputs.properties.fabric_loader_version,
                mysticlib_version: inputs.properties.mysticlib_version
    }
}

shadowJar {
    configurations = [project.configurations.shadowBundle]
    archiveClassifier = 'dev-shadow'
}

remapJar {
    inputFile.set shadowJar.archiveFile
}
